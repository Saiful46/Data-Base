CREATE TABLE COMPANY
  (MOB_NUM          NUMBER(15),
   LOCATION        VARCHAR2(20),
   PRIMARY KEY (MOB_NUM)
);


CREATE TABLE CUSTOMER
  (CUST_NUM         NUMBER(15),
   NID           NUMBER(20),
   NAME           VARCHAR2(15),
   SEX           VARCHAR2(7),
   STAFF_NAME    VARCHAR2(15),  
   PRIMARY KEY (nid),
   FOREIGN KEY (cust_num) REFERENCES COMPANY (mob_num)
);

CREATE TABLE DATA 
	(ID_NUM               NUMBER(15),
	BALANCE_tk             NUMBER(5),
	SMS                    NUMBER(5),
	INERNET_mb             NUMBER(7),
    FOREIGN KEY (id_num) REFERENCEs COMPANY (mob_num)	
);


CREATE TABLE staff (
	staff_id    int,
	Name	Varchar2 (10),
        PRIMARY KEY (staff_id)
);

CREATE TABLE joining (
	emp_id    int,	
	viva_date date,
        join_date date,
 FOREIGN KEY (emp_id) REFERENCES staff (staff_id)

);
